```python
import json
import re

class SimpleTokenizer:
    def __init__(self, vocab_file="vocab_map.json"):
        with open(vocab_file) as f:
            self.vocab = json.load(f)
        self.unk_id = self.vocab.get("<unk>", 0)
        # Simple whitespace + punctuation splitting (not true BPE but works for short prompts)
        self.pat = re.compile(r"\w+|[^\w\s]")
    
    def encode(self, text):
        # Fallback: split into words/chars and map each to id if exists
        tokens = []
        for match in self.pat.finditer(text.lower()):
            word = match.group(0)
            if word in self.vocab:
                tokens.append(self.vocab[word])
            else:
                # Char fallback
                for ch in word:
                    if ch in self.vocab:
                        tokens.append(self.vocab[ch])
                    else:
                        tokens.append(self.unk_id)
        return tokens
    
    def decode(self, ids):
        inv = {v:k for k,v in self.vocab.items()}
        return "".join(inv.get(i, "�") for i in ids)
```

## 3️⃣ Fragment inference engine (pure NumPy, uses sliced weights)

Save as `fragment_engine.py`:

```python
import numpy as np
import json
import os
from tokenizer import SimpleTokenizer

class FragmentLM:
    def __init__(self, fragment_dir="model_fragment", vocab_file="vocab_map.json"):
        with open(f"{fragment_dir}/config.json") as f:
            cfg = json.load(f)
        self.n_embd = cfg['n_embd']
        self.n_head = cfg['n_head']
        self.n_layer = cfg['n_layer']
        self.vocab_size = cfg['vocab_size']
        
        # Load weights
        self.weights = {}
        for fname in os.listdir(fragment_dir):
            if fname.endswith(".npy"):
                key = fname.replace("_", ".").replace(".npy", "")
                self.weights[key] = np.load(f"{fragment_dir}/{fname}")
        
        # Positional embeddings (sinusoidal)
        self.pos_emb = self._sinusoidal_positions(1024)
        
        # Tokenizer
        self.tokenizer = SimpleTokenizer(vocab_file)
        
        # Embedding matrix (from first layer)
        self.embed = self.weights.get("transformer_wte_weight", 
                                      np.random.randn(self.vocab_size, self.n_embd) * 0.02)
    
    def _sinusoidal_positions(self, max_len):
        pe = np.zeros((max_len, self.n_embd))
        for pos in range(max_len):
            for i in range(0, self.n_embd, 2):
                pe[pos, i] = np.sin(pos / (10000 ** (i / self.n_embd)))
                if i+1 < self.n_embd:
                    pe[pos, i+1] = np.cos(pos / (10000 ** ((i+1) / self.n_embd)))
        return pe
    
    def _attention(self, q, k, v):
        # Simplified without mask for speed
        d_k = q.shape[-1]
        scores = np.einsum("qhd,khd->qhk", q, k) / np.sqrt(d_k)
        attn = np.exp(scores - scores.max(axis=-1, keepdims=True))
        attn /= attn.sum(axis=-1, keepdims=True)
        return np.einsum("qhk,khd->qhd", attn, v)
    
    def _layer_norm(self, x, gamma, beta, eps=1e-5):
        mean = x.mean(axis=-1, keepdims=True)
        var = x.var(axis=-1, keepdims=True)
        return (x - mean) / np.sqrt(var + eps) * gamma + beta
    
    def _feed_forward(self, x, fc1_w, fc1_b, fc2_w, fc2_b):
        h = np.maximum(0, x @ fc1_w.T + fc1_b)
        return h @ fc2_w.T + fc2_b
    
    def generate(self, prompt, max_tokens=40, temperature=0.9):
        input_ids = self.tokenizer.encode(prompt)
        generated = list(input_ids)
        
        for _ in range(max_tokens):
            inp = generated[-512:]
            seq_len = len(inp)
            # Embed + pos
            x = np.array([self.embed[i] for i in inp])
            x = x + self.pos_emb[:seq_len]
            
            for l in range(self.n_layer):
                # Attention block
                ln1 = self._layer_norm(x,
                    self.weights[f"transformer_h_{l}_ln_1_weight"],
                    self.weights[f"transformer_h_{l}_ln_1_bias"])
                q = ln1 @ self.weights[f"transformer_h_{l}_attn_q_proj_weight"].T
                k = ln1 @ self.weights[f"transformer_h_{l}_attn_k_proj_weight"].T
                v = ln1 @ self.weights[f"transformer_h_{l}_attn_v_proj_weight"].T
                head_dim = self.n_embd // self.n_head
                q = q.reshape(seq_len, self.n_head, head_dim)
                k = k.reshape(seq_len, self.n_head, head_dim)
                v = v.reshape(seq_len, self.n_head, head_dim)
                attn_out = self._attention(q, k, v)
                attn_out = attn_out.reshape(seq_len, self.n_embd)
                attn_proj = attn_out @ self.weights[f"transformer_h_{l}_attn_out_proj_weight"].T
                x = x + attn_proj
                
                # FFN block
                ln2 = self._layer_norm(x,
                    self.weights[f"transformer_h_{l}_ln_2_weight"],
                    self.weights[f"transformer_h_{l}_ln_2_bias"])
                ff_out = self._feed_forward(ln2,
                    self.weights[f"transformer_h_{l}_mlp_fc1_weight"],
                    self.weights[f"transformer_h_{l}_mlp_fc1_bias"],
                    self.weights[f"transformer_h_{l}_mlp_fc2_weight"],
                    self.weights[f"transformer_h_{l}_mlp_fc2_bias"])
                x = x + ff_out
            
            # Final layer norm
            x = self._layer_norm(x, self.weights["transformer_ln_f_weight"], self.weights["transformer_ln_f_bias"])
            logits = x[-1] @ self.embed.T
            # Temperature sampling
            probs = np.exp(logits / temperature)
            probs /= probs.sum()
            next_id = np.random.choice(len(probs), p=probs)
            generated.append(next_id)
            if next_id == self.tokenizer.vocab.get('.', 0) and len(generated) > 10:
                break
        
        return self.tokenizer.decode(generated)
```

```python
import subprocess
import tempfile
import os
from flask import Flask, render_template, Response, request, jsonify
from fragment_engine import FragmentLM

app = Flask(__name__)

# Load the fragment once
fragment = FragmentLM()

def text_to_pulsar_audio(text):
    with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as f_raw:
        raw_wav = f_raw.name
    subprocess.run(["espeak", "-v", "en+m3", "-s", "14", "-p", "-550", "-a", "180",
                    text, "--stdout"], stdout=open(raw_wav, "wb"), stderr=subprocess.DEVNULL)
    
    with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as f_out:
        out_wav = f_out.name
    
    subprocess.run(["sox", raw_wav, out_wav,
                    "overdrive", "20", "30",
                    "echo", "0.8", "0.7", "100", "0.3",
                    "reverb", "50", "50", "100", "0.4",
                    "tremolo", "4", "0.7",
                    "sinc", "100-3000",
                    "gain", "-2",
                    "channels", "2"], stderr=subprocess.DEVNULL)
    os.unlink(raw_wav)
    return out_wav

@app.route("/")
def index():
    return render_template("index.html")

@app.route("/generate", methods=["POST"])
def generate():
    prompt = request.json.get("prompt", "Speak from the quantum void.")
    try:
        # Generate text from the 1/4 fragment
        fragment_text = fragment.generate(prompt, max_tokens=45, temperature=1.3)
        audio_file = text_to_pulsar_audio(fragment_text)
        def generate_audio():
            with open(audio_file, "rb") as f:
                yield from f
            os.unlink(audio_file)
        return Response(generate_audio(), mimetype="audio/wav")
    except Exception as e:
        return jsonify({"error": str(e)}), 500

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000, debug=False)
```

#### `self_heal.py`

```python
import os
import hashlib
import json
import schedule
import time
import subprocess

BACKUP_FILE = "xor_backup.json"

def hash_file(path):
    with open(path, "rb") as f:
        return hashlib.sha256(f.read()).hexdigest()

def backup_all():
    data = {}
    for root, _, files in os.walk("."):
        for f in files:
            if f.endswith((".py", ".sol", ".js", ".html")):
                full = os.path.join(root, f)
                data[full] = hash_file(full)
    with open(BACKUP_FILE, "w") as b:
        json.dump(data, b)

def self_heal():
    if not os.path.exists(BACKUP_FILE):
        backup_all()
        return
    with open(BACKUP_FILE) as b:
        saved = json.load(b)
    for path, old_hash in saved.items():
        if not os.path.exists(path):
            print(f"⚠️ Missing {path} – restoring from backup")
            subprocess.run(["git", "checkout", path])
        elif hash_file(path) != old_hash:
            print(f"🔧 Corrupted {path} – repairing")
            subprocess.run(["git", "checkout", path])
    # Also restart services
    subprocess.run(["pkill", "-f", "app.py"], stderr=subprocess.DEVNULL)
    subprocess.run(["python3", "app.py", "&"])

if __name__ == "__main__":
    schedule.every(6).hours.do(self_heal)
    while True:
        schedule.run_pending()
        time.sleep(60)
```

```python
import hashlib

class GaslessToken:
    def __init__(self):
        self.balances = {}  # address -> int
        self.nonce = 0

    def transfer(self, from_addr, to_addr, amount, signature):
        # Signature is XOR of (from, to, amount, nonce)
        msg = f"{from_addr}{to_addr}{amount}{self.nonce}".encode()
        expected = hashlib.sha256(msg).digest()
        if signature == expected:
            if self.balances.get(from_addr, 0) >= amount:
                self.balances[from_addr] -= amount
                self.balances[to_addr] = self.balances.get(to_addr, 0) + amount
                self.nonce += 1
                return True
        return False
```
